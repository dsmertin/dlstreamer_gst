271  void clip_normalized_rect(double &x, double &y, double &w, double &h) {
272  if (!((x >= 0) && (y >= 0) && (w >= 0) && (h >= 0) && (x + w <= 1) && (y + h <= 1))) {
273  GST_DEBUG("ROI coordinates x=[%.5f, %.5f], y=[%.5f, %.5f] are out of range [0,1] and will be clipped", x,
274  x + w, y, y + h);
275 
276  x = (x < 0) ? 0 : (x > 1) ? 1 : x;
277  y = (y < 0) ? 0 : (y > 1) ? 1 : y;
278  w = (w < 0) ? 0 : (w > 1 - x) ? 1 - x : w;
279  h = (h < 0) ? 0 : (h > 1 - y) ? 1 - y : h;
280  }
